function init () {
    var numbers = [ 1, 2, 4, 8, 16, 32 ];

    return {
        addEntry: function () {
            numbers.push( Math.pow(2, numbers.length) );
        },
        get: function (n) {
            if( n < numbers.length ) {
                return numbers[n];
            } else {
                console.err("Nice try!");
            }
        }
    };
}

var powersOfTwo = init();

console.log( powersOfTwo.get(5) ); // 32

powersOfTwo.addEntry();
console.log( powersOfTwo.get(6) ); // 64
